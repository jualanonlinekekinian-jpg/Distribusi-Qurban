  <script>
    var namaPengguna = "";
    var dataTerpilih = null;
    var dataSementara = null;
    var intervalProses; // Variabel untuk pembaruan otomatis

    // ✅ FUNGSI FULL SCREEN
    function toggleFullScreen() {
      if (!document.fullscreenElement) {
        document.documentElement.requestFullscreen().catch(err => {
          console.log(`Error: ${err.message}`);
        });
      } else {
        if (document.exitFullscreen) {
          document.exitFullscreen();
        }
      }
    }

    // ✅ FUNGSI GANTI MODE & SAMAKAN KEDUA TOGGLE
    const toggleLogin = document.getElementById('modeToggleLogin');
    const toggleMain  = document.getElementById('modeToggleMain');

    function setMode(isDark) {
      if (isDark) {
        document.body.classList.remove('light-mode');
        localStorage.setItem('modePref', 'dark');
      } else {
        document.body.classList.add('light-mode');
        localStorage.setItem('modePref', 'light');
      }
      // Samakan posisi sakelar
      toggleLogin.checked = isDark;
      toggleMain.checked  = isDark;
    }

    // Jalankan saat sakelar diubah
    toggleLogin.addEventListener('change', () => setMode(toggleLogin.checked));
    toggleMain.addEventListener('change', () => setMode(toggleMain.checked));

    // Cek preferensi saat buka halaman
    window.onload = function() {
      const simpanMode = localStorage.getItem('modePref');
      if(simpanMode === 'light') {
        setMode(false);
      } else {
        setMode(true); // Default Dark Mode
      }
    };

    window.addEventListener("DOMContentLoaded", function() {
      document.getElementById("kataSearch").addEventListener("keyup", function(e) {
        if (e.key === "Enter") cariData();
      });
    });

    function loginAkun() {
      var user = document.getElementById("user").value.trim();
      var pass = document.getElementById("pass").value.trim();
      if (!user || !pass) { tampilkanPopupTidakDitemukan(); return; }

      google.script.run
        .withSuccessHandler(function(res) {
          if (res.status) {
            namaPengguna = res.nama;
            document.getElementById("namaPetugas").innerText = "Petugas: " + namaPengguna;
            document.getElementById("halamanLogin").style.display = "none";
            document.getElementById("halamanUtama").style.display = "flex";
            toggleFullScreen();
            muatProgres();
            // ✅ MULAI PEMBARUAN OTOMATIS SETIAP 5 DETIK
            intervalProses = setInterval(muatProgres, 5000);
          } else {
            tampilkanPopupTidakDitemukan();
          }
        })
        .withFailureHandler(function() { tampilkanPopupTidakDitemukan(); })
        .cekLogin(user, pass);
    }

    function keluarAkun() {
      if (confirm("Yakin ingin keluar?")) {
        // ✅ HENTIKAN PEMBARUAN OTOMATIS SAAT KELUAR
        clearInterval(intervalProses);
        document.getElementById("halamanUtama").style.display = "none";
        document.getElementById("halamanLogin").style.display = "flex";
        toggleFullScreen();
        document.getElementById("user").value = "";
        document.getElementById("pass").value = "";
      }
    }

    /* ── PROGRES DENGAN WARNA KEKINIAN ── */
    function muatProgres() {
      google.script.run.withSuccessHandler(function(data) {
        var html = "";
        data.forEach(function(item) {
          let kelasWarna, warnaBar;
          if (item.persen < 50) {
            kelasWarna = "grad-kurang";
            warnaBar = "#FFB3B3";
          } else if (item.persen < 100) {
            kelasWarna = "grad-setengah";
            warnaBar = "#FFE0B2";
          } else {
            kelasWarna = "grad-ok";
            warnaBar = "#B7F0D0";
          }

          html += `<div class="progress-card ${kelasWarna}">
            <h4>${item.rt}</h4>
            <div class="angka">${item.sudah} / ${item.total}</div>
            <div class="persen">${item.persen}%</div>
            <div class="progress-bar-bg">
              <div class="progress-bar-fill" style="width:${item.persen}%; background:${warnaBar}"></div>
            </div>
          </div>`;
        });
        document.getElementById("daftarProgres").innerHTML = html;
      }).hitungProgres();
    }

    function prosesKode(kode) {
      google.script.run.withSuccessHandler(function(res) {
        var el  = document.getElementById("hasilProses");
        var tek = document.getElementById("teksHasil");
        el.style.display = "block";

        if (res.hasil === "sukses") {
          if(document.body.classList.contains('light-mode')) {
            el.style.background = "#f0fdf4"; el.style.color = "#065f46";
          } else {
            el.style.background = "rgba(22, 163, 74, 0.15)"; el.style.color = "#4ADE80";
          }
          tek.innerHTML = "✅ " + res.pesan + "<br><strong>" + res.data.nama + "</strong> — " + res.data.rt;
          bersihkan();
          setTimeout(function() { el.style.display = "none"; }, 2500);

        } else if (res.hasil === "sudah") {
          if(document.body.classList.contains('light-mode')) {
            el.style.background = "#fffbeb"; el.style.color = "#92400e";
          } else {
            el.style.background = "rgba(234, 120, 0, 0.15)"; el.style.color = "#FFB347";
          }
          tek.innerHTML = "⚠️ " + res.pesan + "<br><strong>" + res.data.nama + "</strong> — " + res.data.rt;
          bersihkan();
          setTimeout(function() { el.style.display = "none"; }, 2500);

        } else {
          if(document.body.classList.contains('light-mode')) {
            el.style.background = "#fef2f2"; el.style.color = "#991b1b";
          } else {
            el.style.background = "rgba(245, 63, 63, 0.15)"; el.style.color = "#FF8080";
          }
          tek.innerHTML = "❌ " + res.pesan;
          bersihkan();
          setTimeout(function() { el.style.display = "none"; }, 2500);
        }
      }).prosesData(kode, namaPengguna);
    }

    function cariData() {
      var kunci = document.getElementById("kataSearch").value.trim();
      if (!kunci) { tampilkanPopupTidakDitemukan(); return; }
      cekStatusData(kunci);
    }

    function cekStatusData(kunci) {
      google.script.run.withSuccessHandler(function(hasil) {
        if (hasil.length === 0) { tampilkanPopupTidakDitemukan(); return; }

        var html = "";
        hasil.forEach(function(item) {
          var sudah = item.status === "SUDAH DIAMBIL";
          var badgeStyle = sudah
            ? "font-size:11px;font-weight:700;padding:4px 10px;border-radius:10px;background:rgba(22, 163, 74, 0.15);color:#4ADE80;white-space:nowrap;flex-shrink:0"
            : "font-size:11px;font-weight:700;padding:4px 10px;border-radius:10px;background:rgba(245, 63, 63, 0.15);color:#FF8080;white-space:nowrap;flex-shrink:0";

          html += '<div class="kartu-cari">'
            + '<div style="min-width:0;overflow:hidden">'
            + '<div style="font-weight:700;color:var(--btn-color);font-size:14px;white-space:nowrap;overflow:hidden;text-overflow:ellipsis">' + item.id_qr + '</div>'
            + '<div style="font-size:13px;color:var(--text-color);white-space:nowrap;overflow:hidden;text-overflow:ellipsis;margin-top:2px">' + item.nama + '</div>'
            + '<div style="font-size:11px;color:var(--text-muted);margin-top:2px">' + item.rt + '</div>'
            + '</div>'
            + '<span style="' + badgeStyle + '">' + (sudah ? "✅ Diambil" : "⏳ Belum") + '</span>'
            + '</div>';
        });
        document.getElementById("hasilCari").innerHTML = html;

        var item = hasil[0];
        setTimeout(function() {
          if (item.status === "SUDAH DIAMBIL") {
            document.getElementById("popupSudahDiambil").style.display = "flex";
          } else {
            // ✅ MENGISI DATA KE POPUP KONFIRMASI
            document.getElementById("konfirm_id_qr").innerText = item.id_qr;
            document.getElementById("konfirm_nama").innerText = item.nama;
            document.getElementById("konfirm_rt").innerText = item.rt;
            
            dataTerpilih = item.id_qr;
            document.getElementById("popupKonfirmasi").style.display = "flex";
          }
        }, 200);

      }).cariData(kunci);
    }
